```python
# 금액 숫자형식으로 변환, 날짜 변환
def valuechange(data):
    data['금액'] = data['거래금액(만원)'].str.replace(' ','')
    data['금액'] = data['금액'].str.replace(',','')
    data['금액'] = data['금액'].astype('int')
    data['계약년월']= pd.to_datetime(data['계약년월'], format='%Y%m')
```


```python
# 층구분
def floor_dist(data):
    data['층구분'] = 0
    
    for i in range(len(data['층'])):
        if data['층'][i] == 1:
            data['층구분'][i] = '1층'
        elif data['층'][i] <= 5:
            data['층구분'][i] = '저층'
        elif data['층'][i] <= 16:
            data['층구분'][i] = '중층'
        else :
            data['층구분'][i] = '고층'
    
```
